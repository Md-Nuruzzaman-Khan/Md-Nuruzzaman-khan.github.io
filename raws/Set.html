    #1.Enter the number of Set(s)
    #2.Enter element of number 1 set(separated by ",")
    #  Enter element of number 2 set
    #       ................
    #  Enter element of number n set
    #3.Enter the Set method (Example: Union,Intersection,Difference,Number of Elements)
    
    SetList=[]
    Newst=[]
    
    n=int(input("Enter the number of Set(s): "))
    
    for x in range(n):
        vv=input("Enter elements of number {} Set: ".format(x+1))
        values=vv.split(",")
        SetList.append(set(values))
    
    mtd1=input("Enter the Set method: ")
    mtd=mtd1.lower().strip()
    if mtd[0]=="u": # Union
        for y1 in range(n):
            Newst+=SetList[y1]
        print("Union = ",set(Newst))
    
    elif mtd[0]=="i": # Intersection
        re=set(SetList[0])
        for j in range(n):
            temp=set(SetList[j])
            for y2 in range(n):
                temp.intersection_update(set(SetList[y2]))
            re.intersection_update(temp)
        if re==set():
            print("Intersection = {}")
        else:
            print("Intersection = ",re)
    
    elif mtd[0]=="d": # Difference
        differ=set(SetList[0]).difference(set(SetList[1]))
        if differ==set():
            print("Difference = {}")
        else:
            print("Difference = ",differ)
       
    elif mtd[0]=="n" or mtd==[0]=="e":
        length=len(set(SetList[0]))
        print("Number of elements = {}".format(length))
    
    else:
        print("Please enter a valid Set method")
